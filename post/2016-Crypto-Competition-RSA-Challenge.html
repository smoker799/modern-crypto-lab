2016 全国高校密码数学挑战赛 RSA 破译赛题解析

2016 年全国高校密码数学挑战赛中的 RSA 破译题，给出了 21 帧 RSA 密文数据，每帧包含模数 N、公钥指数 e 和密文 c。我们需要利用多种 RSA 攻击方法还原明文，并拼接成完整句子。

题目背景与数据格式

每帧数据为 768 字节十六进制串，分为三部分：

  • 前 256 字节：密文 c
  • 中间 256 字节：公钥指数 e
  • 后 256 字节：模数 N

明文结构为 128 字节：前 8 字节标志位 + 4 字节序号 + 8 字节 ASCII 明文片段，通过序号可拼接出完整句子。

攻击方法一览

  • 公共模数攻击（Frame0/4）
  • 共享素因子攻击（Frame1/18）
  • Håstad 广播攻击（e=3 和 e=5 两组）
  • Pollard p-1 分解（Frame2/6/19）
  • Fermat 分解（Frame10/14）
  • LCG 弱随机数攻击（Frame5/9/13/17）

完整实现代码

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
RSA 加密体制破译 - 2016 密码竞赛题解
"""

import binascii
import math
import time
from collections import defaultdict
from itertools import combinations

# ========== 基础数论函数 ==========
def egcd(a, b):
    if a == 0:
        return (b, 0, 1)
    else:
        g, y, x = egcd(b % a, a)
        return (g, x - (b // a) * y, y)

def modinv(a, m):
    g, x, y = egcd(a, m)
    if g != 1:
        raise Exception('模逆不存在')
    return x % m

def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

def isqrt(n):
    return int(math.isqrt(n))

def integer_nth_root(n, k):
    if n < 0:
        return (None, False)
    if k == 2:
        x = isqrt(n)
        return (x, x * x == n)
    low, high = 0, n
    while low <= high:
        mid = (low + high) // 2
        mid_pow = pow(mid, k)
        if mid_pow == n:
            return (mid, True)
        elif mid_pow < n:
            low = mid + 1
        else:
            high = mid - 1
    return (high, False)

# ========== 中国剩余定理 ==========
def chinese_remainder(moduli, remainders):
    total = 0
    prod = 1
    for m in moduli:
        prod *= m
    for n_i, a_i in zip(moduli, remainders):
        p = prod // n_i
        total += a_i * modinv(p, n_i) * p
    return total % prod

# ========== 解析明文块 ==========
def parse_plain_block(m_int):
    m_hex = hex(m_int)[2:].rstrip('L')
    m_hex = m_hex.zfill(128)
    flag = m_hex[:16]
    seq_hex = m_hex[16:24] if len(m_hex) >= 24 else "00000000"
    seq = int(seq_hex, 16)
    last_16 = m_hex[-16:]
    try:
        plain = binascii.unhexlify(last_16).decode('ascii')
        return plain, seq, flag
    except:
        return None, seq, flag

# ========== 1. 公共模数攻击 ==========
def common_modulus_attack(frames):
    results = []
    n_groups = defaultdict(list)
    for frame in frames:
        n_groups[frame['N']].append(frame)
    for N, group in n_groups.items():
        if len(group) < 2:
            continue
        f1, f2 = group[0], group[1]
        g, a, b = egcd(f1['e'], f2['e'])
        if g != 1:
            continue
        try:
            if a < 0:
                c1 = modinv(f1['c'], N)
                a = -a
            else:
                c1 = f1['c']
            if b < 0:
                c2 = modinv(f2['c'], N)
                b = -b
            else:
                c2 = f2['c']
            m = (pow(c1, a, N) * pow(c2, b, N)) % N
            plain, seq, _ = parse_plain_block(m)
            if plain:
                results.append({'frame_id': f1['id'], 'seq': seq, 'plain': plain, 'method': 'common_modulus'})
                results.append({'frame_id': f2['id'], 'seq': seq, 'plain': plain, 'method': 'common_modulus'})
        except:
            pass
    return results

# ========== 2. 共享素因子攻击 ==========
def shared_factor_attack(frames):
    results = []
    for i in range(len(frames)):
        for j in range(i + 1, len(frames)):
            f1, f2 = frames[i], frames[j]
            p = gcd(f1['N'], f2['N'])
            if p == 1 or p == f1['N']:
                continue
            # 分解 Frame1
            q1 = f1['N'] // p
            phi1 = (p - 1) * (q1 - 1)
            try:
                d1 = modinv(f1['e'], phi1)
                m1 = pow(f1['c'], d1, f1['N'])
                plain1, seq1, _ = parse_plain_block(m1)
                if plain1:
                    results.append({'frame_id': f1['id'], 'seq': seq1, 'plain': plain1, 'method': f'shared_factor(Frame{j})'})
            except:
                pass
            # 分解 Frame2
            q2 = f2['N'] // p
            phi2 = (p - 1) * (q2 - 1)
            try:
                d2 = modinv(f2['e'], phi2)
                m2 = pow(f2['c'], d2, f2['N'])
                plain2, seq2, _ = parse_plain_block(m2)
                if plain2:
                    results.append({'frame_id': f2['id'], 'seq': seq2, 'plain': plain2, 'method': f'shared_factor(Frame{i})'})
            except:
                pass
    return results

# ========== 3. Håstad 广播攻击 ==========
def broadcast_attack(frames, e):
    results = []
    e_groups = defaultdict(list)
    for frame in frames:
        e_groups[frame['e']].append(frame)
    for exp, group in e_groups.items():
        if exp != e or len(group) < e:
            continue
        for combo in combinations(group, e):
            n_list = [f['N'] for f in combo]
            c_list = [f['c'] for f in combo]
            try:
                m_e = chinese_remainder(n_list, c_list)
                m, exact = integer_nth_root(m_e, e)
                if exact:
                    plain, seq, _ = parse_plain_block(m)
                    if plain:
                        for f in combo:
                            results.append({'frame_id': f['id'], 'seq': seq, 'plain': plain, 'method': f'broadcast_e={e}'})
            except:
                continue
    return results

# ========== 4. Pollard p-1 分解 ==========
def get_primes(limit):
    sieve = [True] * (limit + 1)
    sieve[0] = sieve[1] = False
    for i in range(2, int(limit ** 0.5) + 1):
        if sieve[i]:
            for j in range(i * i, limit + 1, i):
                sieve[j] = False
    return [i for i, is_p in enumerate(sieve) if is_p]

def pollard_p_minus_1(n, bound=200000):
    a = 2
    primes = get_primes(bound)
    for prime in primes:
        power = prime
        while power * prime <= bound:
            power *= prime
        a = pow(a, power, n)
        d = gcd(a - 1, n)
        if 1 < d < n:
            return d
    return None

def pollard_attack(frame, bound=200000):
    p = pollard_p_minus_1(frame['N'], bound)
    if p is None:
        return None
    q = frame['N'] // p
    phi = (p - 1) * (q - 1)
    try:
        d = modinv(frame['e'], phi)
        m = pow(frame['c'], d, frame['N'])
        plain, seq, _ = parse_plain_block(m)
        if plain:
            return {'frame_id': frame['id'], 'seq': seq, 'plain': plain, 'method': 'pollard_p_minus_1'}
    except:
        return None
    return None

# ========== 5. Fermat 分解 ==========
def fermat_factorization(n, max_offset=100000):
    a = isqrt(n)
    if a * a < n:
        a += 1
    for offset in range(max_offset):
        a_test = a + offset
        b2 = a_test * a_test - n
        if b2 < 0:
            continue
        b = isqrt(b2)
        if b * b == b2:
            return a_test - b, a_test + b
    return None, None

def fermat_attack(frame, max_offset=100000):
    p, q = fermat_factorization(frame['N'], max_offset)
    if p is None:
        return None
    phi = (p - 1) * (q - 1)
    try:
        d = modinv(frame['e'], phi)
        m = pow(frame['c'], d, frame['N'])
        plain, seq, _ = parse_plain_block(m)
        if plain:
            return {'frame_id': frame['id'], 'seq': seq, 'plain': plain, 'method': 'fermat_factorization'}
    except:
        return None
    return None
